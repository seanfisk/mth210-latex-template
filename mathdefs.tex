% Use an short regular arrow instead of a double arrow for implications.
\renewcommand{\implies}{\rightarrow}

% Use a centered dot for multiplication.
\newcommand{\mult}{\cdot}

% Number sets
\newcommand{\N}{\mathbb{N}}
\newcommand{\Z}{\mathbb{Z}}
\newcommand{\Q}{\mathbb{Q}}
\newcommand{\R}{\mathbb{R}}

% Abbreviations.
\newcommand{\st}{\text{ s.t. }}

% Boolean operations
% All caps so we don't override existing commands.
\newcommand{\AND}{\wedge}
\newcommand{\OR}{\vee}

% Create new theorem commands.
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{corollary}[theorem]{Corollary}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{conjecture}[theorem]{Conjecture}
\newtheorem*{definition}{Definition}

% By default, the QED symbol is an empty square aligned right. We want it to be a filled square aligned right.
\renewcommand{\qed}{\hfill\ensuremath{\blacksquare}}
